extension LinkedList {
    func value(at index: Int) -> String? {
        guard index >= 0 else { return "valor no v√°lido"}
        var i = 0
        var node = head
        while node != nil && i < index {
            node = node!.next
            i = i+1
        }
        return node?.value
    }
}
